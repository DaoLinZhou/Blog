信号signal

安装信号: 两种方式

signal(int sig, void (*func)(int));

int sigaction(int sig, 
              const struct sigaction* act,
             struct sigaction* oact);

struct sigaction{
    // 处理函数0
    void(*sa_handler)(int);

    // 处理函数1
    void (*sa_sigaction)(int, siginfo_t*, void*);

    // 掩码, 当执行action时, 如果再有相应的信号出现的时候, 是否相应
    sigset_t sa_mask;

    // 用于判断使用那个处理函数 sa_handler/sa_sigaction
    int sa_flags; // 0 则使用处理函数0, 1 则使用1

    // 设置为NULL, 一般不用
    void(*sa_restorer)(void);
};

使用第一种方式安装信号

#include <iostream>
#include <signal.h>
#include <unistd.h>

void sighandle(int sig){
    std::cout << "receved signal: " << sig << std::endl;
}

int main(int argc, char* argv[]){
    signal(SIGINT, sighandle);
    signal(SIGQUIT, sighandle);
    signal(SIGHUP, sighandle);

    // stop the program and wating for signal 
    pause();
    return 0;
}

使用第二种方式安装信号

#include <iostream>

#include <signal.h>
#include <unistd.h>

void sighandler(int sig){
    std::cout << "received sig: " << sig << std::endl;
}

// define struct as sigaction and type name is sig
typedef struct sigaction sig;

int main(int argc, char* argv[]){
    // 如果不使用typedef的话也可以这样定义
    // struct sigaction act, oact;
    sig act, oact;

    // 绑定第一个处理函数
    act.sa_handler = sighandler;
    // 使用掩码, 处理期间屏蔽所有的信号
    sigfillset(&act.sa_mask);
    // 使用sighandler
    act.sa_flags = 0;

    sigaction(SIGINT, &act, &oact);

    pause();
    return 0;
}

使用fork方式

步骤:

  1. fork创建一个子进程, 父进程退出, 子进程成为孤儿进程, 被init进程接管
  2. 虽然父进程退出, 但相关信息还保留着, sessionId依然标记原来的父进程是多少在那个session中, 所以要调用 setsid 建立新的进程会话
  3. 将当前工作目录切换到根目录
  4. 后台程序没有标准输出和标准出错的, 将标准输入, 输出, 出错重定向到 /dev/null
#include <iostream>

#include <fcntl.h>

/// 为了使用fork(), stop()
#include <unistd.h>
#include <stdlib.h>

void daemonize(){
    int fd;
    pid_t pid;
    // <1>, 记录fork的pid, 如果<0说明fork失败
    if((pid = fork()) < 0){
        std::cout << "can't creat subprocess" << std::endl;
        /// pid 如果为0 说明是子进程
        /// 父进程(当前执行的程序) 非零
    }else if(pid != 0){
        exit(0);
    }

    // <2>
    setsid();

    // <3>, 改变工作目录
    if(chdir("/") < 0){
        std::cout << "can't change dir" << std::endl;
        exit(-1);
    }

    /// <4>, 重定向标准流
    fd = open("/dev/null", O_RDWR);
    dup2(fd, STDIN_FILENO);
    dup2(fd, STDOUT_FILENO);
    dup2(fd, STDERR_FILENO);

    return;
}

int main(int argc, char* argv[]){
    daemonize();
    while(1){
        // 防止运行太快,  cpu
        sleep(1);
    }
}

使用 daemon API 的方式

daemon也是按照fork的四个步骤进行实现的

#include <iostream>

#include <unistd.h>
#include <stdlib.h>

int main(int argc, char* argv[]){

    // 第一个参数: 0 代表切换到根目录下, 非0 代表不切换
    // 第二个参数: 0 代表重定向标准流到/dev/null
    // 运行成功返回0 否则返回-1
    if(daemon(0, 0) == -1){
        // 如果不成功
        std::cout << "error" << std::endl;
        exit(-1);
    }

    while(1){
        sleep(1);
    }    

    return 0;
}

网络编程基础

七大步骤

  1. 创建socket, 指定使用tcp协议 (相当于打开了一个文件 / 创建了一个文件句柄)
  2. 将socket 与 地址和端口绑定 ( 在操作系统底层就会把事件抛给fd(文件描述符), 即这个socket)
  3. 监听端口, 不然虽然事件抛过去了, 但如果不见听上层应用依然不知道, 捕获不到时间
  4. 创建新的socket, 作数据的收发, 而不再继续占用监听的socket
  5. 使用recv接收数据
  6. 使用send发生数据
  7. 使用close发送数据

TCP常见套接字选项

  • SO_REUSEADDR 端口处于 WAIT_TIME 依然可以启动 (关闭socket后会有一段time wait时间, 直到收到finish, 设置它时候就不用等待一段时间了, 即可以启动另外一个程序依然使用这个地址和端口, 对于服务端来说非常有用)
  • SO_RCVBUF 接收的buffer的大小
  • SO_SNDBUF 发送的buffer的大小, 设置太大会有延迟

重要的结构体

struct sockaddr_in
{
    // 使用的协议 例如: IPV4
    sa_family_t        sin_family;
    // 设置端口
    uint16_t        sin_port;
    // 设置ip
    struct in_addr    sin_addr;
    // 填充位
    char            sin_zero[8];
}

// ip地址就是这个结构体
struct in_addr
{
    // 一个整型
    in_addr_t        s_addr;
}

// 由于这个对开发任意不太友好, 所以才有了上面sockaddr_in
// 要通过强制类型转换欺骗编译器
struct sockaddr
{
    // 使用的协议 例如: IPV4
    sa_family_t        sin_family;
    char            sin_zero[14];
}

TCP服务端实现例子

#include <iostream>
#include <sys/socket.h>
#include <netinet/in.h>
#include <unistd.h>
#include <string.h>

#define PORT 8111
#define MESSAGE_LEN 1024


int main(int agrc, char* argv[]){

    // 缓冲区大小 
    int backlog = 10;

    // 记录返回值
    int ret = -1;

    // 设置选项是否起作用
    int on = 1;

    // 文件资源描述符
    int socket_fd, accept_fd;

    char in_buff[MESSAGE_LEN] = {0,};

    // 本地地址和客户端的地址
    struct sockaddr_in localaddr, remoteaddr;

    // AF_INET IPV4 协议, SOCK_STREAM TCP协议
    socket_fd = socket(AF_INET, SOCK_STREAM, 0);
    if ( socket_fd == -1 ) {
        std::cout << "Failed to create socket!" << std::endl;
        exit(-1);
    }

    ret = setsockopt( socket_fd,
               SOL_SOCKET, SO_REUSEADDR, 
               &on, sizeof(on) );

    if (ret == -1) {
        std::cout << "Failed to set socket option!" << std::endl;
    }

    localaddr.sin_family = AF_INET;
    localaddr.sin_port = PORT;
    // INADDR_ANY = 0, all ip addr will listen to this port
    localaddr.sin_addr.s_addr = INADDR_ANY;
    // 其余字节填充为0
    bzero(&(localaddr.sin_zero), 8);

    // 绑定地址和端口, 通过强转欺骗编译器
    ret = bind(socket_fd, (struct sockaddr *)&localaddr, sizeof(struct sockaddr));

    if(ret == -1){
        std::cout << "Failed to bind addr!" << std::endl;
        exit(-1);
    }

    // 监听, 并发的时候, 同一时间(毫秒级)只能处理一个, 其余的都要放在缓冲区中, backlog设置缓冲区的大小
    ret = listen(socket_fd, backlog);

    if(ret == -1){
        std::cout << "Failed to listen socket!" << std::endl;
        exit(-1);
    }

    for(;;){
        socklen_t addr_len = sizeof(struct sockaddr);
        accept_fd = accept(socket_fd,
                        (struct sockaddr *) &remoteaddr,
                        &addr_len);
        for(;;){
            ret = recv(accept_fd, (void *) in_buff, MESSAGE_LEN, 0);
            if(ret == 0){
                break;
            }
            std::cout << "recv : " << in_buff << std::endl;
            send(accept_fd, (void *) in_buff, MESSAGE_LEN, 0);
        }
        close(accept_fd);
    }
    close(socket_fd);
    return 0;
}

TCP客户端实现例子

#include <iostream>
#include <sys/socket.h>
#include <sys/types.h>
#include <string.h>
#include <stdio.h>
#include <netinet/in.h>
#include <unistd.h>
#include <arpa/inet.h>

#define PORT 8111
#define MESSAGE_LEN 1024

int main(int argc, char* argv[]){

    int socket_fd;
    int ret = -1;
    struct sockaddr_in serveraddr;
    char sendbuf[MESSAGE_LEN] = {0,};
    char recvbuf[MESSAGE_LEN] = {0,};


    socket_fd = socket(AF_INET, SOCK_STREAM, 0);
    if(socket_fd < 0){
        std::cout << "Failed to created socket!" << std::endl;
        exit(-1);
    }

    serveraddr.sin_family = AF_INET;
    serveraddr.sin_port   = PORT;
    serveraddr.sin_addr.s_addr = inet_addr("127.0.0.1"); 

    ret = connect(socket_fd, 
                 (struct sockaddr *) &serveraddr, 
                 sizeof(struct sockaddr));
    if(ret < 0){
        std::cout << "Faile to connect server!" << std::endl;
        exit(-1);
    }

    for(;;){
        memset(sendbuf, 0, MESSAGE_LEN);
        fgets(sendbuf, MESSAGE_LEN, stdin);
        ret = send(socket_fd, sendbuf, strlen(sendbuf), 0);
        if(ret <= 0){
            std::cout << "Faile to send data!" << std::endl;
            break;
        }

        if(strcmp(sendbuf, "quit") == 0){
            break;
        }

        ret = recv(socket_fd, recvbuf, MESSAGE_LEN, 0);
        recvbuf[ret] = '\0';
        std::cout << "recvbuf : " << recvbuf << std::endl;
    }

    close(socket_fd);
    return 0;
}

UDP服务端实现例子

#include <iostream>
#include <sys/socket.h>
#include <netinet/in.h>
#include <unistd.h>
#include <string.h>
#include <sys/types.h>


#define PORT 8111
#define MESSAGE_LEN 1024

int main(int agrc, char* argv[]){

    int ret;
    int socket_fd;
    int on = 1;
    char buf[MESSAGE_LEN] = {0};

    sockaddr_in localaddr, recvaddr;
    int len = sizeof(sockaddr);
    socket_fd = socket(AF_INET, SOCK_DGRAM, 0);

    if(socket_fd == -1){
        std::cout << "Failed to create socket!" << std::endl;
        exit(-1);
    }

    ret = setsockopt(socket_fd, SOL_SOCKET, SO_REUSEADDR, 
               &on, sizeof(on));
    if (ret == -1) {
        std::cout << "Failed to set socket option!" << std::endl;
    }

    localaddr.sin_family = AF_INET;
    localaddr.sin_port = PORT;
    // INADDR_ANY = 0, all ip addr will listen to this port
    localaddr.sin_addr.s_addr = INADDR_ANY;
    // 其余字节填充为0
    bzero(&(localaddr.sin_zero), 8);

    ret = bind(socket_fd, (sockaddr *) &localaddr, sizeof(struct sockaddr));
    if(ret == -1){
        std::cout << "Failed to bind addr!" << std::endl;
        exit(-1);
    }

    recvaddr.sin_family = AF_INET;
    recvaddr.sin_port = PORT;
    localaddr.sin_addr.s_addr = INADDR_ANY;
    bzero(&(localaddr.sin_zero), 8);

    for(;;){
        memset(buf, 0, MESSAGE_LEN);
        ret = recvfrom(socket_fd, (void *) buf, MESSAGE_LEN, 0, 
                        (sockaddr *) &recvaddr,(socklen_t *) &len);
        if ( ret == 0 ) {
            break;
        }
        std::cout << "recv : " << buf << std::endl;
        sendto(socket_fd, (void *) buf, MESSAGE_LEN, 0, 
                        (sockaddr *) &recvaddr, len);
    }
    close(socket_fd);
    return 0;
}
