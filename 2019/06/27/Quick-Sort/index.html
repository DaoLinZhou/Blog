快速排序是21世纪最伟大的算法之一

许多系统内部就是使用快速排序来进行排序的(有点绕口??)

本文从最简单的快速排序开始, 演进写出性能较好的快速排序

普通快排

代码复杂程度: 简单

时间复杂度: O(nlogn)

空间复杂度: O(1)

稳定性: 不稳定

缺点:

  1. 对近乎有序的数组会退化成 n^2 级别的算法
  2. 基准元素的选择有规律(不随机)
  3. 没有考虑相等元素的情况

思想: 随机选一个数, 对数组进行整理, 左边都比这个数小, 右边都比他大, 这个数就在他排好序后应该在的位置

// 以第一个元素为基础, 左边都比它小, 右边都比他大
template <typename T>
int __partition(T arr[], int l,int r){
    int p = l;
    for(int i = l+1; i <= r; i++)
        if(arr[i] < arr[l])
            swap(arr[++p], arr[i]);
    swap(arr[l], arr[p]);
    return p;
}

template <typename T>
void __quickSort(T arr[], int l, int r){
//    if(l>=r)
//        return;
    if(r-l > 15){        // 优化: 插入排序 arr[l...r]    这步优化会大大提升性能
        insertionSort(arr, l, r);
        return;
    }
    int p = __partition(arr, l, r);
    __quickSort(arr, l, p-1);
    __quickSort(arr, p+1, r);
}

template <typename T>
void quickSort(T arr[], int n){
    __quickSort(arr, 0, n-1);
}

优化版普通快速排序

随机算法, 有极大的概率效率非常高, 极低的概率退化为 n^2 级别(近乎为0)

template <typename T>
int __partition(T arr[], int l,int r){
    //随机选择一个元素, 和第一个元素互换, 之后依然以第一个元素作为基准
    swap(arr[l], arr[rand()%(r-l+1)+l]);
    T e = arr[l];
    int p = l;
    for(int i = l+1; i <= r; i++)
        if(arr[i] < arr[l])
            swap(arr[++p], arr[i]);
    swap(arr[l], arr[p]);
    return p;
}

template <typename T>
void __quickSort(T arr[], int l, int r){
//    if(l>=r)
//        return;
    if(r-l > 15){
        insertionSort(arr, l, r);
        return;
    }
    int p = __partition(arr, l, r);
    __quickSort(arr, l, p-1);
    __quickSort(arr, p+1, r);
}

template <typename T>
void quickSort(T arr[], int n){
    srand(time(NULL));
    __quickSort(arr, 0, n-1);
}

双路快排

代码复杂程度: 中等

时间复杂度: O(nlogn)

空间复杂度: O(1)

稳定性: 不稳定

思想: 和普通快排大致一样, 但要把等于标定点的元素均匀分布在大于和小于的部分

swap i 和 j

template <typename T>
int __partition2(T arr[], int l,int r){
    swap(arr[l], arr[rand()%(r-l+1)+l]);
    int p = l+1, q = r;
    T e = arr[l];
    while (true){
        while (p<=r && arr[p] < e) p++;
        while (q>=l+1 && arr[q] > e) q--;
        if(p > q) break;
        swap(arr[p], arr[q]);
        p++;
        q--;
    }
    swap(arr[l], arr[q]);
    return q;
}

template <typename T>
void __quickSort2(T arr[], int l, int r){
//    if(l>=r)
//        return;
    if(r-l > 15){
        insertionSort(arr, l, r);
        return;
    }
    int p = __partition2(arr, l, r);
    __quickSort2(arr, l, p-1);
    __quickSort2(arr, p+1, r);
}

template <typename T>
void quickSort2(T arr[], int n){
    srand(time(NULL));
    __quickSort2(arr, 0, n-1);
}

三路快排

代码复杂程度: 中等

时间复杂度: O(nlogn)

空间复杂度: O(1)

稳定性: 不稳定

特点: 处理有重复值的数组优势非常大, 系统级别一般用它. 而没有处理重复值数组的速度也可以保证

思想: 三条路线, 小于,等于,大于, 进行排序

p : less后下一个元素位置

q : equal后下一个元素位置

k : more 前一个元素

template <typename T>
void __quickSort3(T arr[], int l, int r){
    if(l>=r)
        return;
//    if(r-l > 15){
//        insertionSort(arr, l, r);
//        return;
//    }
    swap(arr[l], arr[rand()%(r-l+1)+l]);
    T e = arr[l];
    //这里的起始位置非常重要, 要明白起始位置代表的是什么
    int p=l+1, q=l+1, k=r;
    while (q <= k){
        if(arr[q] == e)    //如果等于
            q++;
        else if(arr[q] < e) //如果小于
            swap(arr[p++], arr[q++]);
        else    //如果大于
            swap(arr[q], arr[k--]);
    }
    //p-1 指向less中最后一个元素, 而不应该是p
    swap(arr[l], arr[p-1]);

    //递归
    __quickSort3(arr, l, p-2);
    __quickSort3(arr, q, r);
}

// 比较Merge Sort和双路快速排序和三路快排三种排序算法的性能效率
// 对于包含有大量重复数据的数组, 三路快排有巨大的优势
// 对于一般性的随机数组和近乎有序的数组, 三路快排的效率虽然不是最优的, 但是是在非常可以接受的范围里
// 因此, 在一些语言中, 三路快排是默认的语言库函数中使用的排序算法。比如Java:)
template <typename T>
void quickSort3(T arr[], int n){
    srand(time(NULL));
    __quickSort3(arr, 0, n-1);
}

非递归三路快排

如果想把一个递归算法, 改写成非递归算法, 最好的辅助数据结构就是栈(stack)

大体思路没变, 只是把递归操作改为入栈操作

但由于使用了栈, 所以会消耗额外的数据空间


using namespace std;

template <typename T>
void __partition(T arr[], stack<int> *st){
    //先取出右边界, 再取出左边界
    int r = st->top();
    st->pop();
    int l = st->top();
    st->pop();

    if(l>=r)
        return;

    swap(arr[l], arr[rand()%(r-l+1)+l]);
    T v = arr[l];

    int lt = l+1, k = l+1, gt = r;
    while(k <= gt){
        if(arr[k] == v){
            k++;
        }else if(arr[k] < v){
            swap(arr[k++], arr[lt++]);
        }else{
            swap(arr[k], arr[gt--]);
        }
    }
    swap(arr[l], arr[lt-1]);

    //分别入栈
    st->push(l);
    st->push(lt-2);
    st->push(gt+1);
    st->push(r);
}

template <typename T>
void quickSortLoop(T arr[], int n){
    srand(time(0));
    stack<int> *st = new stack<int>();
    //先放左边界, 再放右边界
    st->push(0);
    st->push(n-1);
    while(!st->empty()){
        __partition(arr, st);
    }
}

总结:

mergeSort 与 quickSort 都使用了分治算法, 而这两者又代表了分治算法的两种情况

mergeSort : 分简单, 归并难

quickSort : 分难,归并简单(不用归并, 只要分就可以)


获取数组中第N大的元素

通过快速排序, 每一次会把一个元素放大它排好序应该在的位置, 通过与这个位置比较我们能直到应该是向前处理还是向后处理. 直到找到我们想要的元素

#ifndef ALGORITHM_NTHNUMBER_H
#define ALGORITHM_NTHNUMBER_H

#include <iostream>
#include <ctime>

using namespace std;

template <typename T>
int __quickSort(T arr[], int l, int r, int rank){
    if(l >= r)
        return arr[l];
    swap(arr[l], arr[rand()%(r-l)+l]);
    T v = arr[l];
    int lt = l+1, gt = r, q = l+1;
    while(q <= gt){
        if(arr[q] == v)
            q++;
        else if(arr[q] > v)
            swap(arr[q], arr[gt--]);
        else
            swap(arr[q++], arr[lt++]);
    }
    swap(arr[l], arr[--lt]);

    if(rank >= lt && rank < q) {
        //cout << rank << " " << lt << ", " << gt << endl << "number:" << arr[lt] << endl;
        return arr[lt];
    }else if(rank <= (lt-1))
        return __quickSort(arr, l, lt - 1, rank);
    else
        return __quickSort(arr, q, r, rank);
}

template <typename T>
// n        数组长度
// rank     希望获取第 rank 大的元素 从 1 到 n
int getNthNumber(T arr[], int n, int rank){
    srand(time(NULL));
    return __quickSort(arr, 0, n-1, n-rank);
}

#endif //ALGORITHM_NTHNUMBER_H
