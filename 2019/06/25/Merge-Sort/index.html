归并排序分为 自顶向下 和 自底向上 两种实现

分别使用 递归 和 迭代, 性能都是 O(nlogn)

自底向上(循环)的归并排序，是比自顶向下(递归)的归并排序要快的


自顶向下 递归实现

代码复杂程度: 中等

时间复杂度: O(nlogn)

空间复杂度: O(n)

稳定性: 稳定

思想: 递归, 将两个排好序的数组归并为一个有序数组

视频演示: 

// 插入排序 arr[l...r], 为归并排序进行优化
template <typename T>
void insertionSort(T arr[], int l, int r){
    for(int i=l+1; i<=r; i++){
        T e = arr[i];
        int j;
        for( j = i; j > l && e < arr[j-1]; j -- )
            arr[j] = arr[j-1];
        arr[j] = e;
    }
}
template <typename T>
// 将arr[l...mid] 和 arr[mid+1...r] 两个部分进行归并
void __merge(T arr[], int l, int mid, int r){
    // 开辟辅助空间
    T *aux = new T[r-l+1];

    // 辅助空间的元素和要merge的元素一样
    for( int i = l; i <= r; i++ )
        aux[i-l] = arr[i];

    // 初始化，i指向左半部分的起始索引位置l；j指向右半部分起始索引位置mid+1
    int i = l, j = mid+1;
    for( int k = l ; k <= r; k ++ ){
        if( i > mid ){  // 如果左半部分元素已经全部处理完毕
            arr[k] = aux[j-l]; j ++;
        }
        else if( j > r ){  // 如果右半部分元素已经全部处理完毕
            arr[k] = aux[i-l]; i ++;
        }
        else if( aux[i-l] < aux[j-l] ) {  // 左半部分所指元素 < 右半部分所指元素
            arr[k] = aux[i-l]; i ++;
        }
        else{  // 左半部分所指元素 >= 右半部分所指元素
            arr[k] = aux[j-l]; j ++;
        }
    }
    delete[] aux;
}

// 递归使用归并排序, 对 arr [l...r] 范围进行排序
template <typename T>
void __mergeSort(T arr[], int l, int r){
    //if( l >= r )
    //    return;
    //优化: 当小于一定范围时, 近乎有序的可能会变大, 用插入排序进行优化
    if( r-l <= 15 ){
        insertionSort(arr, l, r);
        return;
    }
    int mid = (r-l)/2+l;
    __mergeSort(arr, l, mid);
    __mergeSort(arr, mid+1, r);

    //优化: 如果两个排好序的数组一个最后一个元素小于另一个的第一个元素
    // 那么这两个本身就是有序的, 反之 则要merge
    if(arr[mid] >= arr[mid+1])
        __merge(arr, l, mid, r);
}

template <typename T>
void mergeSort(T arr[], int n){
    __mergeSort(arr, 0, n-1);
}

自底向上 循环实现

代码复杂程度: 中等

时间复杂度: O(nlogn)

空间复杂度: O(n)

稳定性: 稳定

思想: 遍历, 把当前这组元素和相邻的一组元素进行merge

例如: sz = 1 时, 就是把奇数位元素和他后一个元素比较, 交换位置

//没有使用数组的一个特性, 通过下标直接获取元素, 可以用nlogn的时间为链表排序
template <typename T>
void mergeSortB2U(T arr[], int n){
    //sz 为步长
    for(int sz=1; sz <= n; sz += sz)
        // 每次 merge i 到 i+sz-1 和 i+sz 到 i+2*sz-1
        // i+sz<n 是确保 mid+1 存在, 即至少有两组数据进行merge
        // 因为 i+sz+sz-1 可能会越界, 所以取它和n-1中较小的一个
        for(int i=0; i+sz < n; i += sz+sz)
            __merge(arr, i, i+sz-1, min(i+sz+sz-1,n-1));
}

根据实现的不同, 两种代码具体的执行方式也有所不同. 但相同的一点就是merge的操作.

然而, 这两种归并排序的实现都避不开一个问题, 那就是空间复杂度, 他们都开辟了O(n)的空间


拓展应用:

求数组中的逆序对

从数组中随机取两个数, 若他们的顺序是逆序的, 则他们构成了一组逆序对

通常一个数组中逆序对的个数 代表了数组的混乱程度

//
// Created by Daolin on 2019/6/9.
//
#include <iostream>
#include "SortTestHelper.h"

using namespace std;

template <typename T>
long long __merge(T arr[], int l, int mid ,int r){
    T* aux = new T[r-l+1];
    for(int i=l; i<=r; i++)
        aux[i-l] = arr[i];

    long long count = 0;
    int p=l, q = mid+1;
    for(int i = l; i <= r; i++){
        if(p > mid){
            arr[i] = aux[q - l];
            q++;
        }else if(q > r){
            arr[i] = aux[p - l];
            p++;
        }else if(aux[p - l] > aux[q - l]){    //左边大于右边
            arr[i] = aux[q - l];
            count += mid-p+1;                //逆序对计数
            q++;
        }else{                                //右边大于左边
            arr[i] = aux[p - l];
            p++;
        }
    }

    delete[] aux;
    return count;
}

template <typename T>
long long merge(T arr[], int l, int r){
    if (l>=r)
        return 0;
    int mid = (r-l)/2 + l;
    long long count_1 = merge(arr, l, mid);
    long long count_2 = merge(arr, mid+1, r);
    long long count = __merge(arr, l, mid, r);

    return count_1+count_2+count;
}

template <typename T>
long long sortAndGetReverseOrder(T arr[], int n){
    return merge(arr, 0, n-1);
}


int main(){

    int n = 1000;
    // 生成长度为1000的近乎有序的数组, 比较简单代码就不放上来了
    int* arr = SortTestHelper::generateNearlyOrderedArray(n,1);

    long long count = sortAndGetReverseOrder(arr, n);
    cout << count << endl;

    return 0;
}
