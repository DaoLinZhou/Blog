之前也说过, 一个图中所有节点不一定都是连通的

那么, 在某些情况下就会需要计算图的连通分量

注意: 由于图的作用不同, 所以图的实现也会相应地调整, 下面的计算连通分量的算法, 仅适用于上一章的两种实现

实现

#ifndef ALGORITHM_COMPONENT_H
#define ALGORITHM_COMPONENT_H

#include <iostream>
#include <cassert>
using namespace std;

template <typename Graph>
class Component{
private:
    Graph &G;
    bool* visited;    // 记录节点是否被访问过
    int ccount;     // 记录有多少个连通分量
    int* id;        // 每个节点都有一个id, id一致的节点是连通的

    void dfs(int v){        // 深度优先遍历 deep first search
        visited[v] = true;

        /** 
         * 这里设计的比较巧妙, 因为ccount只有在深度遍历完成后才自增
         * 所以同一次深度遍历时ccount是同一个值, 而深度遍历是遍历当前节点的所有连通节点
         * 所以连通节点的id是一致的
         */
        id[v] = ccount;        

        // 注意, 这里必须要有typename, 告诉编译器adjIterator 是一个类型而不是方法
        typename Graph::adjIterator adj(G, v);    

        // i为v的每个相邻节点
        for(int i = adj.begin(); !adj.isEnd(); i = adj.next())
            if( !visited[i] )    // 只有当节点没有被访问过的时候才递归遍历
                dfs(i);
    }

public:
    // 求一个图的连通分量不需要什么花里胡哨的参数, 只要图本身就可以了
    Component(Graph &graph):G(graph){    
        visited = new bool[G.V()];
        id = new int[G.V()];
        ccount = 0;
        // 初始化
        for(int i = 0; i < G.V(); i++) {
            visited[i] = false;
            id[i] = -1;
        }
        // 解析图的连通分量
        for(int i = 0; i < G.V(); i++){
            if(!visited[i]){
                dfs(i);        // 被深度遍历到的节点visited会被标为true, 所以不会二次访问
                ccount ++;    // 只有遍历后才增加连通分量的总数
            }
        }
    }

    ~Component(){
        delete[] visited;
        delete[] id;
    }

    int count(){
        return ccount;
    }

    bool isConnected(int v, int w){
        assert(v >= 0 && v < G.V());
        assert(w >= 0 && w < G.V());
        return id[v] == id[w];
    }

};

#endif //ALGORITHM_COMPONENT_H
