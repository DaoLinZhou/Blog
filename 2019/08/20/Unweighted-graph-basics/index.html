一个图主要是由节点和边来组成

图主要用于表示两个事物之间的关系:

  1. 交通运输

  2. 社交网络

  3. 工作安排

  4. 程序状态执行

    … …

图的分类

图根据边的方向性分为两类:

​ 有向图(Directed Graph)

​ 无向图(Undirected Graph)

无向图可以看作是特殊的有向图, 就是所有节点都双向的


根据边的权重也可以分为两类

​ 无权图(Unweighted Graph)

​ 有权图(Weighted Graph)

边可以包含一些信息


图的连通性

图不一定非要是联通的(也可以是移动/电信的), 从一个点到另外一个点不一定有一条路


图根据实现的不同可以有 自环边(自己指向自己) 和 平行边(从一点到另一点有多条边).


图的实现

图通常情况下有两种实现

  1. 邻接矩阵 - 用于稠密图
  2. 邻接表 - 用于稀疏图

可以看出, 当每个节点和大部分其他节点产生关系时(稠密图), 比较适合邻接矩阵

当每个节点只和部分节点产生联系(稀疏图), 则更适合邻接表


完全图: 特殊的稠密图, 每个节点和其他所有节点产生联系


邻接矩阵 - 稠密图

#ifndef ALGORITHM_DENSEGRAPH_H
#define ALGORITHM_DENSEGRAPH_H

#include <iostream>
#include <vector>
#include <cassert>

using namespace std;

// 稠密图 - 邻接矩阵
class DenseGraph{
private:
    int n, m;   // n 为节点数, m为边数
    bool directed;
    vector<vector<bool>> g;

public:
    DenseGraph(int n, bool directed){
        this->n = n;
        this->m = 0;
        this->directed = directed;
        for(int i = 0; i < n; i ++){
            g.push_back(vector<bool>(n , false));
        }
    }

    ~DenseGraph() {

    }

    int V(){return n;}
    int E(){return m;}

    void addEdge(int v, int w){
        assert(v >= 0 && v < n);
        assert(w >= 0 && w < n);

        if(hasEdge(v, w))
            return;

        g[v][w] = true;
        if(!directed)
            g[w][v] = true;
        m++;
    }

    bool hasEdge(int v, int w){
        assert(v >= 0 && v < n);
        assert(w >= 0 && w < n);
        return g[v][w];
    }

    void show(){
        for(int i = 0; i < n; i++){
            for(int j = 0; j < n; j++)
                cout << g[i][j] << "\t";
            cout << endl;
        }
    }

    // 相邻边的迭代器
    class adjIterator{
    private:
        DenseGraph &G;
        int v;
        int index;
    public:
        adjIterator(DenseGraph &graph, int v):G(graph){
            this->v = v;
            this->index = -1;
        }

        int begin(){
            index = -1;
            return next();
        }

        int next(){
            for(index += 1; index < G.V(); index++)
                if(G.g[v][index])
                    return index;
            return -1;
        }

        bool isEnd(){
            return index >= G.V();
        }
    };

};

#endif

邻接表 - 稀疏图

#ifndef ALGORITHM_SPARSEGRAPH_H
#define ALGORITHM_SPARSEGRAPH_H

#include <iostream>
#include <vector>
#include <cassert>

using namespace std;

// 稀疏图 - 邻接表
class SparseGraph{
private:
    int n, m;   // 点数 边数
    bool direct;
    vector<vector<int>> g;

public:
    SparseGraph( int n, bool directed ){
        this->n = n;
        this->m = m;
        this->direct = directed;
        for(int i=0; i<n; i++)
            g.push_back(vector<int>());
    }

    ~SparseGraph() {}

    int V(){ return n; }
    int E(){ return m; }

    void addEdge(int v, int w){
        assert(v >= 0 && v < n);
        assert(w >= 0 && w < n);

        g[v].push_back(w);
        if( v != w &&  !direct )
            g[w].push_back(v);
        m++;
    }

    bool hasEdge(int v, int w){
        assert(v >= 0 && v < n);
        assert(w >= 0 && w < n);
        for(int i=0; i<g[v].size(); i++)
            if(g[v][i] == w)
                return true;
        return false;
    }

    void show(){
        for(int i=0; i<n; i++){
            cout << "vertex " << i << ":\t";
            for(int j = 0; j < g[i].size(); j++)
                cout << g[i][j] << "\t";
            cout << endl;
        }
    }

    // 相邻边的迭代器
    class adjIterator{
    private:
        SparseGraph &G;
        int v;
        int index;
    public:
        adjIterator(SparseGraph &graph, int v):G(graph){
            this->v = v;
            this->index = 0;
        }

        int begin(){
            index = 0;
            if(G.g[v].size())
                return G.g[v][0];
            return -1;
        }

        int next(){
            index++;
            if(index < G.g[v].size())
                return G.g[v][index];
            return -1;
        }

        bool isEnd(){
            return index >= G.g[v].size();
        }
    };

};


#endif

注意:

  1. 我们要保持稀疏图和稠密图的接口一样

  2. 当我们要遍历图中某一个节点的相邻节点时, 虽然可以有其他方式, 但从设计, 扩展性, 接口一致性的角度思考, 通过内部类 adjIterator 来实现更为优雅.
