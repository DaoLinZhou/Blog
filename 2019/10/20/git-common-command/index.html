git 常用命令

配置

git config --global user.name 'your_name'
git config --global user.email 'your_email@domain.com'
git config --list --global         # 显示配置信息

提交

git add .        # 跟踪当前目录下的文件, 放出暂存区
git commit -m "first commit"    # 提交

查看

git status        # 查看暂存区的状态
git log            # 查看过去commit的状态
git log --oneline     # 简单查看
git log -n4     # 查看最近的4次
git log --graph    # 以图形的方式查看commit的父子关系


git diff --cached     # 查看暂存区的和HEAD中的差别
git diff         # 默认查看工作区(还没commit的)和暂存区中的差别

# 可以指定多个文件
git diff -- 文件名     # 查看指定文件名的工作区和暂存区的区别, -- 可以省略
git diff --cached 文件名     # 查看指定文件暂存区和HEAD中的差别

git diff <hash> <hash> [-- <file>]  # 查看两个commit中文件的差异, 可以指定看哪个文件

分支

git branch -av    # 查看本地分支
git checkout -b temp 21c243 # 基于给定hash创建分支, 分支名为temp, 同时切换到该分支
git checkout (branchname) # 切换分支
git branch -d (branchname/hash) # 删除分支
git branch -D (branchname/hash) # 强行删除分支, 当-d不好用时可以用它

修改

git mv oldname newname # 更改文件名

git commit --amend     # 修改最近一次的commit message

git rebase -i hash # 修改以name/hash为根commit, 可以修改, 压缩 等等

git rm <file>     # 删除文件, 工作区和暂存区都删除

恢复

git reset HEAD <file>        # 暂存区回复成HEAD一样

git checkout -- <file>        # 把工作区文件恢复成和暂存区一样

git reset --hard <commit>        # 把头指针指向指定commit, 同时把暂存区和工作区回复到这个commit一样的状态

git stash

stash可以把当前工作现场“保存”起来,等以后恢复现场后继续工作 

git stash             # 执行存储
git stash list        # 查看一共有哪些存储
git stash apply     # 应用某个存储, 但不会从列表中删除, 默认使用第一个存储
git stash apply stash@{2} # 使用第二个存储

git stash pop        # 和apply一样, 但会从列表中删除 也可以用stash@{}来指定
